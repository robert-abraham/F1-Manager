Generate 100 marks from 40-99, write it to a file, read that file, Calculate the min,max,range, and avg

from random import randint

#Create file and write 100 marks ranged from 40-99
file =  open("MarkOutput.txt", "w")
for i in range(100):
    file.write(str(randint(40,99))+"\n")
file.close()


#Read file Filter out \n and convert list to integer items
with open("MarkOutput.txt", "r") as file:
    marks = " ".join(file.readlines()).replace('\n','').split()
    marks = list(map(int, marks)) #This converts the string list into an integer list
file.close()


#Calculate Values
avg_marks = (sum(marks) /len(marks))
max_mark = max(marks)
min_mark = min(marks)
mark_range = max_mark-min_mark

#Print Values to console
print(f"Average Marks: {avg_marks}")  
print(f"Maximum Mark: {max_mark}")
print(f"Minimum Mark: {min_mark}")
print(f"Range of Marks: {mark_range}")

#Write Values to Stat File
with open("ExamStats.txt", "w") as statfile:
    statfile.write(f"Average Marks: {avg_marks}\n")
    statfile.write(f"Maximum Mark: {max_mark}\n")
    statfile.write(f"Minimum Mark: {min_mark}\n")
    statfile.write(f"Range of Marks: {mark_range}")
statfile.close()
